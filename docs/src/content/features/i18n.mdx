## Usage

### Setup

The `TranslationsProvider` wraps the entire app in `layout.tsx`:

```typescript
// src/app/layout.tsx
import { TranslationsProvider } from '@/lib/i18n/translations-context';

export default function RootLayout({ children }) {
  return (
    <html>
      <body>
        <TranslationsProvider>
          {children}
        </TranslationsProvider>
      </body>
    </html>
  );
}
```

### Using Translations

```typescript
import { useTranslations } from '@/lib/i18n/translations-context';

function MyComponent() {
  const { t, language, setLanguage, dir } = useTranslations();
  
  return (
    <div dir={dir}>
      <h1>{t('hero.title')}</h1>
      <button onClick={() => setLanguage('es')}>
        {t('language.es')}
      </button>
    </div>
  );
}
```

### Hook API

```typescript
const {
  language,        // Current language code: 'en' | 'es' | 'fr' | 'zh' | 'ar' | 'bn'
  setLanguage,     // Function to change language
  t,               // Translation function: (key: string) => string
  formatDate,      // Date formatter: (date: Date | string) => string
  dir,             // Text direction: 'ltr' | 'rtl'
} = useTranslations();
```

## Adding Translations

### Adding a New Key

1. Open `src/lib/i18n/translations-context.tsx`
2. Add the key to **all** language objects:

```typescript
const translations = {
  en: {
    'new.key': 'English text',
    // ...existing keys
  },
  es: {
    'new.key': 'Spanish text',
    // ...existing keys
  },
  // Repeat for fr, zh, ar, bn
};
```

3. Use in components:

```typescript
<p>{t('new.key')}</p>
```

### Adding a New Language

1. Add language code to type:

```typescript
export type Language = 'en' | 'es' | 'fr' | 'zh' | 'ar' | 'bn' | 'de'; // Add 'de' for German
```

2. Add RTL if applicable:

```typescript
const rtlLanguages: Language[] = ['ar']; // Add language code if RTL
```

3. Add translations object:

```typescript
const translations = {
  // ...existing languages
  de: {
    'nav.home': 'Startseite',
    'nav.about': '√úber mich',
    // ... all keys
  },
};
```

4. Add language name:

```typescript
// In each language object:
'language.de': 'Deutsch',
```

5. Update browser detection:

```typescript
if (['en', 'es', 'fr', 'zh', 'ar', 'bn', 'de'].includes(browserLanguage)) {
  setLanguage(browserLanguage);
}
```

## RTL Support

Arabic (`ar`) automatically switches to right-to-left layout.

### How It Works

```typescript
// In TranslationsProvider
const setLanguage = (lang: Language) => {
  // ...
  document.documentElement.dir = rtlLanguages.includes(lang) ? 'rtl' : 'ltr';
};
```

### Using in Components

```typescript
const { dir } = useTranslations();

<div dir={dir} className="flex gap-4">
  {/* Content flows RTL when Arabic is selected */}
</div>
```

### RTL-Aware CSS

Tailwind automatically flips these utilities in RTL:
- `ml-4` ‚Üí `mr-4`
- `text-left` ‚Üí `text-right`
- `pl-6` ‚Üí `pr-6`

Use logical properties for explicit RTL support:
- `ms-4` (margin-start)
- `me-4` (margin-end)
- `ps-6` (padding-start)
- `pe-6` (padding-end)

## Best Practices

### 1. Always Use Keys, Not Hardcoded Text

```typescript
// ‚ùå Bad
<Button>Save Project</Button>

// ‚úÖ Good
<Button>{t('admin.saveProject')}</Button>
```

### 2. Keep Keys Consistent

Follow the naming convention:
```
section.component.element
```

### 3. Provide Context in Keys

```typescript
// ‚ùå Ambiguous
'button.save': 'Save'

// ‚úÖ Clear context
'project.button.save': 'Save Project'
'settings.button.save': 'Save Settings'
```

### 4. Handle Missing Translations

The `t()` function falls back in this order:
1. Current language translation
2. English translation
3. The key itself

```typescript
// If 'new.key' is missing in Spanish:
t('new.key')
// ‚Üí English translation if available
// ‚Üí 'new.key' if no translation exists
```

### 5. Test RTL Layout

Always test Arabic language to verify RTL layout works correctly.

### 6. Keep Translations Synchronized

When adding/modifying keys, update **all** language objects to prevent missing translations.

## Language Switcher Component

Example implementation:

```typescript
import { useTranslations, Language } from '@/lib/i18n/translations-context';
import { Button } from '@/components/ui/button';
import { DropdownMenu, DropdownMenuTrigger, DropdownMenuContent, DropdownMenuItem } from '@/components/ui/dropdown-menu';
import { Globe } from 'lucide-react';

const languages: { code: Language; name: string }[] = [
  { code: 'en', name: 'English' },
  { code: 'es', name: 'Espa√±ol' },
  { code: 'fr', name: 'Fran√ßais' },
  { code: 'zh', name: '‰∏≠Êñá' },
  { code: 'ar', name: 'ÿßŸÑÿπÿ±ÿ®Ÿäÿ©' },
  { code: 'bn', name: '‡¶¨‡¶æ‡¶Ç‡¶≤‡¶æ' },
];

function LanguageSwitcher() {
  const { language, setLanguage, t } = useTranslations();
  
  return (
    <DropdownMenu>
      <DropdownMenuTrigger asChild>
        <Button variant="ghost" size="icon">
          <Globe className="h-4 w-4" />
        </Button>
      </DropdownMenuTrigger>
      <DropdownMenuContent>
        {languages.map((lang) => (
          <DropdownMenuItem
            key={lang.code}
            onClick={() => setLanguage(lang.code)}
            className={language === lang.code ? 'bg-accent' : ''}
          >
            {t(`language.${lang.code}`)}
          </DropdownMenuItem>
        ))}
      </DropdownMenuContent>
    </DropdownMenu>
  );
}
```
